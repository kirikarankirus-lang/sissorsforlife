import sys # Sample house listings listings = [ {"address": "1 Sunderland Street", "price": "$350,000", "bedrooms": 3, "bathrooms": 2, "description": "Cozy family home with garden."}, {"address": "4a Marieville Street", "price": "$420,000", "bedrooms": 4, "bathrooms": 3, "description": "Spacious townhouse near city center."} ] def find_house(address_query): results = [] for house in listings: if address_query.lower() in house["address"].lower(): results.append(house) return results def print_listing(house): print(f"Address: {house['address']}") print(f"Price: {house['price']}") print(f"Bedrooms: {house['bedrooms']}") print(f"Bathrooms: {house['bathrooms']}") print(f"Description: {house['description']}") print("-" * 40) def main(): if len(sys.argv) < 2: print("Usage: python listing_script.py
") sys.exit(1) query = " ".join(sys.argv[1:]) matched_houses = find_house(query) if matched_houses: for house in matched_houses: print_listing(house) else: print("No houses found for that address.") if __name__ == "__main__": main()
